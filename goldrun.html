const config = { type: Phaser.AUTO, width: 800, height: 400, parent: 'game-container', physics: { default: 'arcade', arcade: { gravity: { y: 900 }, debug: false } }, scene: { preload: preload, create: create, update: update } }; const game = new Phaser.Game(config); // Variables de Juego let player, obstacles, gasItems, cursors; let lives = 5; let gasCount = 0; let gameSpeed = -300; let isGameOver = false; // Billetera de Quema Oficial (Dead Address) const BURN_WALLET = "0x0000000000000000000000000000000000000000"; const STAKE_AMOUNT = "0.0001"; function preload() { // Assets desde tus CIDs de IPFS (usando gateway de cloudflare para velocidad) this.load.image('background', 'https://cloudflare-ipfs.com/ipfs/bafybeibbjca3baec45r7rd3ggks2i7pcxliv3ntdy3u3s26mffrcwr4kwu'); this.load.image('player', 'https://cloudflare-ipfs.com/ipfs/bafybeid4eh5si4tdy3kpqgtnd4l2z6ehac3rkd4vjmsgdb5gqbs2vcox34'); this.load.image('obstacle', 'https://cloudflare-ipfs.com/ipfs/bafybeiarw4qbftgyh2rzjyw56s4cpjxtlptxtwgasgnu7rovqkvmpjgbgy'); this.load.image('gas', 'https://cloudflare-ipfs.com/ipfs/bafybeihsmtiwinw5w6jcebelg2p6kr64ivuijcmx4dnpuny2m22piqin2y'); // Sonidos (Asegúrate de tener estos archivos o cámbialos por tus links) this.load.audio('jump', 'https://labs.phaser.io/assets/audio/SoundEffects/p-ping.mp3'); this.load.audio('collect', 'https://labs.phaser.io/assets/audio/SoundEffects/pickup.wav'); } function create() { this.add.image(400, 200, 'background').setDisplaySize(800, 400); player = this.physics.add.sprite(100, 300, 'player').setScale(0.5); player.setCollideWorldBounds(true); obstacles = this.physics.add.group(); gasItems = this.physics.add.group(); // UI this.statsText = this.add.text(16, 16, `Vidas: 5 | Gas: 0/200`, { fontSize: '20px', fill: '#fff', backgroundColor: '#000' }); this.msgText = this.add.text(400, 50, '', { fontSize: '24px', fill: '#00ff00' }).setOrigin(0.5); cursors = this.input.keyboard.createCursorKeys(); // Generadores this.time.addEvent({ delay: 2000, callback: spawnObstacle, callbackScope: this, loop: true }); this.time.addEvent({ delay: 3000, callback: spawnGas, callbackScope: this, loop: true }); // Frases de ánimo const frases = ["¡Dale Panga!", "¡ZAARD al frente!", "¡No te detengas!", "¡Casi llegas!"]; this.time.addEvent({ delay: 5000, callback: () => { if(!isGameOver) this.msgText.setText(Phaser.Utils.Array.GetRandom(frases)); }, loop: true }); // Colisiones this.physics.add.overlap(player, obstacles, hitObstacle, null, this); this.physics.add.overlap(player, gasItems, collectGas, null, this); } function update() { if (isGameOver) return; if ((cursors.up.isDown || cursors.space.isDown) && player.body.blocked.down) { player.setVelocityY(-500); this.sound.play('jump'); } // Aumento de velocidad progresivo gameSpeed -= 0.1; } function spawnObstacle() { let obs = obstacles.create(850, 350, 'obstacle').setScale(0.4); obs.setVelocityX(gameSpeed); } function spawnGas() { let g = gasItems.create(850, Phaser.Math.Between(150, 300), 'gas').setScale(0.3); g.body.allowGravity = false; g.setVelocityX(gameSpeed); } function hitObstacle(player, obstacle) { obstacle.destroy(); lives--; this.statsText.setText(`Vidas: ${lives} | Gas: ${gasCount}/200`); this.cameras.main.shake(200, 0.01); if (lives <= 0) endGame(false, this); } function collectGas(player, gas) { gas.destroy(); gasCount++; this.sound.play('collect'); this.statsText.setText(`Vidas: ${lives} | Gas: ${gasCount}/200`); if (gasCount >= 200) endGame(true, this); } function endGame(won, scene) { isGameOver = true; scene.physics.pause(); let resultText = won ? "¡GANASTE 0.0001 ZAARD!" : "PERDISTE. 0.0001 QUEMADO"; let color = won ? "#00ff00" : "#ff0000"; scene.add.text(400, 200, resultText, { fontSize: '40px', fill: color }).setOrigin(0.5); if (!won) { console.log(`Enviando apuesta a la billetera de quema: ${BURN_WALLET}`); // Aquí iría la llamada a tu contrato para ejecutar la quema } }
